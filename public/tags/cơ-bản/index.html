Trong bài viết này, chúng ta sẽ tìm hiểu chi tiết về những tính năng quan trọng nhất của ES6. Let và Const // Let - phạm vi block scope let x = 10; if (true) { let x = 20; // biến mới, khác với x ở ngoài console.log(x); // 20 } console.log(x); // 10 // Const - không thể gán lại giá trị const PI = 3.14159; // PI = 3.14; // Lỗi: không thể gán lại giá trị cho hằng số 2. Arrow Functions // Cú pháp truyền thống function add(a, b) { return a + b; } // Arrow function const add = (a, b) => a + b; // Arrow function với nhiều dòng const multiply = (a, b) => { const result = a * b; return result; }; // Arrow function trong callback const numbers = [1, 2, 3]; const doubled = numbers.map(num => num * 2); 3. Template Literals const name = 'John'; const age = 30; // Cách cũ console.log('Xin chào, tôi là ' + name + ' và tôi ' + age + ' tuổi.'); // Template literals console.log(`Xin chào, tôi là ${name} và tôi ${age} tuổi.`); // Template literals nhiều dòng const html = ` <div> <h1>${name}</h1> <p>Tuổi: ${age}</p> </div> `; 4. Destructuring Assignment // Destructuring Array const colors = ['red', 'green', 'blue']; const [first, second, third] = colors; console.log(first); // red // Destructuring Object const person = { name: 'John', age: 30, city: 'New York' }; const { name, age } = person; console.log(name); // John // Đổi tên biến khi destructuring const { name: fullName, age: years } = person; console.log(fullName); // John 5. Spread và Rest Operators // Spread operator với arrays const arr1 = [1, 2, 3]; const arr2 = [...arr1, 4, 5]; console.log(arr2); // [1, 2, 3, 4, 5] // Spread operator với objects const obj1 = { foo: 'bar', x: 42 }; const obj2 = { foo: 'baz', y: 13 }; const clonedObj = { ...obj1 }; const mergedObj = { ...obj1, ...obj2 }; // Rest parameter const sum = (...numbers) => { return numbers.reduce((acc, cur) => acc + cur, 0); }; console.log(sum(1, 2, 3, 4)); // 10 6.
